struct ListNode* middleNode(struct ListNode* head) {
    if(head == NULL)
        return head;
    //思路：双指针，一个指针走一步一个指针走两步，快指针走到NULL的时候，慢的在中间。
    struct ListNode* slow = head;
    struct ListNode* fast = head;
        
    while(fast != NULL && slow != NULL)
    {
        if(fast->next == NULL){
            return slow;
        }
        fast = fast->next;
        slow = slow->next;
        if(fast->next == NULL){
            return slow;
        }
        fast = fast->next;
    }
    return slow;
}
