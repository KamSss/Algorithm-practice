struct ListNode* removeElements(struct ListNode* head, int val){
    if(head == NULL)
        return NULL;
     struct ListNode* cur = head;
	 struct ListNode* prev = NULL;

	 while (cur != NULL){
		 //值不相等
		 if (cur->val != val){
			 prev = cur;
			 cur = cur->next;
		 }
		 //值相等
		 else{
             cur = cur->next;
             prev->next = cur; 
		 }
     }
        if(head->val == val){
            struct ListNode* newHead = head->next;
            free(head);
            return newHead;
        }
        else{
            return head;
        }
}
